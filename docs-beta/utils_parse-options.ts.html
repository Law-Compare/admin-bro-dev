import { BlogOptionsWithDefault, BlogOptions } from '../types/blog-options.type'
import { DEFAULT_TAG_NAME_COLUMN, ERROR_MESSAGES, REQUIRED_PROPERTIES } from '../constants'

/**
 * It verifies options (throws errors when some properties are missing), and adds default values
 *
 * @param options original options passed to @softwarebrothers/admin-bro-blog
 */
export const parseOptions = (options: BlogOptions): BlogOptionsWithDefault => {
  const { properties = {}, tags } = options
  const missingProperties = REQUIRED_PROPERTIES
    .filter((reqProp) => Object.keys(properties).indexOf(reqProp) < 0)

  if (missingProperties.length) {
    throw new Error(ERROR_MESSAGES.MISSING_PROPERTY_OPTIONS(missingProperties))
  }

  return {
    ...options,
    tags: tags ? {
      ...tags,
      nameColumn: tags.nameColumn || DEFAULT_TAG_NAME_COLUMN,
    } : undefined,
  }
}
